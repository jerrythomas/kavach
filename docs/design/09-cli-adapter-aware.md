# Adapter-Aware Configuration Design

## Overview

Extends the CLI design (from `2026-03-02-cli-package-design.md`) to support adapter-aware configuration. The CLI reads adapter capabilities and adjusts prompts, generation, and validation accordingly.

## Architecture

### Adapter Capabilities Interface

Each adapter package exports a capabilities object:

```typescript
// adapters/<name>/src/capabilities.ts

export interface AdapterCapability {
  data?: boolean
  rpc?: boolean
  logging?: boolean
  magic?: boolean
  oauth?: boolean
  password?: boolean
  passkey?: boolean
}

export interface DDLTemplate {
  table: string
  schema: string
}

export interface AdapterCapabilities {
  name: string
  displayName: string
  supports: AdapterCapability
  ddl?: {
    logging?: DDLTemplate
  }
}
```

```
│                                 │

│ - Use configured table name     │
└─────────────────────────────────┘
```

## Implementation

### 1. Adapter Capabilities Export

Each adapter adds a `capabilities.ts`:

```typescript
// adapters/supabase/src/capabilities.ts
export const capabilities = {
  name: 'supabase',
  displayName: 'Supabase',
  supports: {
    data: true,
    rpc: true,
    logging: true,
    magic: true,
    oauth: true,
    password: true,
    passkey: true
  },
  ddl: {
    logging: {
      table: 'audit_logs',
      getSchema: (tableName: string) => `
        CREATE TABLE ${tableName} (
          id BIGINT GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
          level TEXT NOT NULL,
          message TEXT NOT NULL,
          context JSONB DEFAULT '{}',
          user_id TEXT,
          created_at TIMESTAMPTZ DEFAULT NOW()
        );
        
        -- Enable Row Level Security
        ALTER TABLE ${tableName} ENABLE ROW LEVEL SECURITY;
        
        -- Allow service role full access
        CREATE POLICY "Service role can manage" ON ${tableName}
          FOR ALL USING (auth.role() = 'service_role');
      `
    }
  }
}
```

### 2. CLI Adapter Loading

```typescript
// packages/cli/src/adapters.js
import { capabilities as supabaseCapabilities } from '@kavach/adapter-supabase'
import { capabilities as firebaseCapabilities } from '@kavach/adapter-firebase'
// ... other adapters

const ADAPTER_REGISTRY = {
  supabase: supabaseCapabilities,
  firebase: firebaseCapabilities,
  // ...
}

export function getAdapterCapabilities(adapterName: string) {
  return ADAPTER_REGISTRY[adapterName]
}

export function getSupportedFeatures(capabilities) {
  return Object.entries(capabilities.supports)
    .filter(([_, supported]) => supported)
    .map(([feature]) => feature)
}
```

### 3. Updated Prompts

```typescript
// packages/cli/src/prompts.js
export async function promptAdapterSelection() {
  const adapters = [
    { value: 'supabase', label: 'Supabase', hint: 'Supports: OAuth, Password, Magic, Data, RPC, Logging' },
    { value: 'firebase', label: 'Firebase', hint: 'Supports: OAuth, Password, Magic' },
    { value: 'auth0', label: 'Auth0', hint: 'Supports: OAuth, Password, Magic' },
    { value: 'amplify', label: 'AWS Amplify', hint: 'Supports: OAuth, Password' },
    { value: 'convex', label: 'Convex', hint: 'Supports: OAuth, Password (data via Convex)' }
  ]
  
  return await select('Select auth adapter:', { options: adapters })
}

export async function promptDataRoute(capabilities) {
  if (!capabilities.supports.data) {
    return {
      enabled: false,
      reason: 'Not supported by selected adapter',
      path: null
    }
  }
  
  return await promptDataRouteConfig()
}

export async function promptLoggingConfig(capabilities) {
  if (!capabilities.supports.logging) {
    return {
      enabled: false,
      reason: 'Not supported by selected adapter',
      table: null,
      level: null
    }
  }
  
  return await promptLoggingConfigWithDDL(capabilities.ddl.logging)
}

export async function promptRPCRoute(capabilities) {
  if (!capabilities.supports.rpc) {
    return {
      enabled: false,
      reason: 'Not supported by selected adapter',
      path: null
    }
  }
  
  return await promptRPCRouteConfig()
}
```

### 4. DDL Output

```typescript
// packages/cli/src/ddl.js
export function generateDDLInstructions(capabilities, config) {
  if (!capabilities.ddl?.logging || !config.logging?.enabled) {
    return null
  }
  
  const tableName = config.logging.table || capabilities.ddl.logging.table
  const schema = capabilities.ddl.logging.getSchema(tableName)
  
  return {
    title: 'DDL Required',
    message: `Run the following SQL in your ${capabilities.displayName} dashboard to enable logging:`,
    sql: schema,
    hint: `Table: ${tableName}`
  }
}

export function formatDDLOutput(ddl) {
  if (!ddl) return ''
  
  return `
┌─────────────────────────────────────────────────────────────┐
│  ${ddl.title}                                              │
└─────────────────────────────────────────────────────────────┘

${ddl.message}

\`\`\`sql
${ddl.sql}
\`\`\`

${ddl.hint}
`
}
```

### 5. Updated Config Generation

```typescript
// packages/cli/src/generators.js
export function generateConfigFile(config, capabilities) {
  const supportedFeatures = capabilities.supports
  
  // Only include config for supported features
  const filteredConfig = {
    adapter: config.adapter,
    providers: config.providers,
    cachedLogins: config.cachedLogins,
    routes: config.routes,
    rules: config.rules,
    env: config.env,
    // Only add if supported
    ...(supportedFeatures.data && { dataRoute: config.dataRoute }),
    ...(supportedFeatures.rpc && { rpcRoute: config.rpcRoute }),
    ...(supportedFeatures.logging && { 
      logging: {
        level: config.logging?.level,
        table: config.logging?.table
      }
    })
  }
  
  return `export default ${serializeJS(filteredConfig, 0)}`
}
```

### 6. Config Validation

```typescript
// packages/cli/src/validation.js
export function validateConfig(config, capabilities) {
  const warnings = []
  const errors = []
  
  // Check unsupported features
  if (config.dataRoute && !capabilities.supports.data) {
    warnings.push('Data route configured but not supported by adapter')
  }
  
  if (config.rpcRoute && !capabilities.supports.rpc) {
    warnings.push('RPC route configured but not supported by adapter')
  }
  
  if (config.logging && !capabilities.supports.logging) {
    warnings.push('Logging configured but not supported by adapter')
  }
  
  return { warnings, errors }
}
```

## Testing Strategy

1. **Unit tests** for capability filtering
2. **Integration tests** for each adapter initialization flow
3. **DDL tests** for SQL generation with custom table names
4. **Validation tests** for warning/error generation
